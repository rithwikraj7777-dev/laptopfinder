<script>
let laptops = [
    {name:"Lenovo Legion 5", price:1200, performance:9, battery:6, display:8, weight:5},
    {name:"HP Omen 14", price:1400, performance:9, battery:8, display:8, weight:8},
    {name:"MacBook Pro", price:1600, performance:8, battery:10, display:9, weight:9},
    {name:"ASUS Vivobook", price:800, performance:6, battery:7, display:7, weight:8}
];

function scoreLaptop(laptop){
    return (laptop.performance*0.4 +
            laptop.battery*0.2 +
            laptop.display*0.15 +
            laptop.weight*0.15 +
            (10 - laptop.price/200)*0.1);
}

function findLaptop(){
    let budget = document.getElementById("budget").value;
    let bestLaptop = null;
    let bestScore = 0;

    laptops.forEach(laptop => {
        if(laptop.price <= budget){
            let score = scoreLaptop(laptop);
            if(score > bestScore){
                bestScore = score;
                bestLaptop = laptop;
            }
        }
    });

    if(bestLaptop){
        document.getElementById("result").innerHTML =
            bestLaptop.name + " (Score: " + bestScore.toFixed(1) + ")";
    } else {
        document.getElementById("result").innerHTML = "No laptop found in budget.";
    }
}
</script>
